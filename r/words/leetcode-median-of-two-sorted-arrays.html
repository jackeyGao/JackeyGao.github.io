class Solution(object):
    def findMedianSortedArrays(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: float
        """
        nums = sorted(nums1 + nums2)
        if len(nums) % 2:
            s = len(nums) / 2
        else:
            s = len(nums) / 2 - 1

        e = len(nums) / 2 + 1

        rs = nums[s:e]
        if len(rs) == 1:
            return rs[0]
        else:
            return sum(rs) / float(2)

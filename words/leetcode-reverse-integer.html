class Solution(object):
    def reverse(self, x):
        """
        :type x: int
        :rtype: int
        """
        
        answer = 0
        sign = 1 if x > 0 else -1
        x = abs(x)
        while x > 0:
            answer = answer * 10 + x % 10
            x /= 10
            
            # update 2014-11-10
            if answer.bit_length() >= 32:
                return 0
                
        return sign * answer

class Solution(object):
    def reverse(self, x):
        """
        :type x: int
        :rtype: int
        """
        value = ''.join([ str(abs(x))[i] for i in range(len(str(abs(x))) - 1, -1, -1) ])
        if x > 0:
            number = int(value)
        else:
            number = -int(value)
            
        if number.bit_length() >= 32:
            return 0
        else:
            return number
